    <script>
        // --- 1. LLAVE DE CONEXIÓN A GOOGLE APPS SCRIPT ---
        const URL_API_APPS_SCRIPT = "https://script.google.com/macros/s/AKfycbxlsEEzXTRmk2zpmXPeqhTIqdcxKTmI_J3aZfUMgb33NPfvk_xLcR_Sb6rsyXn6zFTj/exec";

        // --- 2. GESTIÓN DE METAS ---
        let metaSemanal = localStorage.getItem('metaSemanal') || 30000;
        let metaMensual = localStorage.getItem('metaMensual') || 200000;
        let utilidadSemanaActual = 18500; // Valor por defecto
        let utilidadMesActual = 145200;   // Valor por defecto

        function renderizarMetas() {
            document.getElementById('meta-sem-valor').innerText = Number(metaSemanal).toLocaleString();
            document.getElementById('meta-mes-valor').innerText = Number(metaMensual).toLocaleString();

            let pctSemanal = Math.min(Math.round((utilidadSemanaActual / metaSemanal) * 100), 100);
            let pctMensual = Math.min(Math.round((utilidadMesActual / metaMensual) * 100), 100);

            document.getElementById('meta-sem-porcentaje').innerText = pctSemanal + "%";
            document.getElementById('meta-sem-barra').style.width = pctSemanal + "%";
            document.getElementById('meta-sem-restante').innerHTML = utilidadSemanaActual >= metaSemanal ? "¡META SUPERADA! 🏆" : `Faltan <span class="text-white">$${(metaSemanal - utilidadSemanaActual).toLocaleString()}</span>`;

            document.getElementById('meta-mes-porcentaje').innerText = pctMensual + "%";
            document.getElementById('meta-mes-barra').style.width = pctMensual + "%";
            document.getElementById('meta-mes-restante').innerHTML = utilidadMesActual >= metaMensual ? "¡META SUPERADA! 🏆" : `Faltan <span class="text-white">$${(metaMensual - utilidadMesActual).toLocaleString()}</span>`;
        }

        function configurarMetas() {
            let nuevaMetaSem = prompt("Ingresa tu nueva Meta de Utilidad SEMANAL (solo números, sin comas):", metaSemanal);
            let nuevaMetaMes = prompt("Ingresa tu nueva Meta de Utilidad MENSUAL (solo números, sin comas):", metaMensual);

            if(nuevaMetaSem && nuevaMetaMes && !isNaN(nuevaMetaSem) && !isNaN(nuevaMetaMes)) {
                // Mostrar pantalla de carga
                const loader = document.getElementById('loading-dash');
                document.getElementById('loading-text').innerText = "Guardando en Servidor Central...";
                loader.style.display = 'flex';
                loader.style.opacity = '1';

                // Simulación de envío a la nube
                setTimeout(() => {
                    metaSemanal = nuevaMetaSem;
                    metaMensual = nuevaMetaMes;
                    localStorage.setItem('metaSemanal', metaSemanal);
                    localStorage.setItem('metaMensual', metaMensual);
                    renderizarMetas();
                    
                    loader.style.opacity = '0';
                    setTimeout(() => { loader.style.display = 'none'; }, 300);
                }, 1000); 
            }
        }

        // --- 3. CÁLCULOS DE TIEMPO ---
        function getNumeroSemana(d) {
            d = new Date(Date.UTC(d.getFullYear(), d.getMonth(), d.getDate()));
            d.setUTCDate(d.getUTCDate() + 4 - (d.getUTCDay()||7));
            var yearStart = new Date(Date.UTC(d.getUTCFullYear(),0,1));
            return Math.ceil((((d - yearStart)/86400000)+1)/7);
        }

        const hoy = new Date();
        const numSemana = getNumeroSemana(hoy);
        const ayer = new Date(hoy); ayer.setDate(ayer.getDate() - 1);
        
        document.getElementById('fecha-actual').innerText = hoy.toLocaleDateString('es-MX', { weekday: 'long', day: 'numeric', month: 'long' }).toUpperCase();
        document.getElementById('label-semana-header').innerText = `SEM ${numSemana}`;
        document.getElementById('label-ayer').innerText = ayer.toLocaleDateString('es-MX', { weekday: 'short', day: 'numeric' }).toUpperCase();

        // --- 4. MOTOR DE CONEXIÓN A LA NUBE (FETCH) ---

                // 5. INYECTAR LIVE FEED

                // 6. INYECTAR STOCK CRÍTICO

                        <div class="text-[10px] font-bold text-slate-200">${i.p} <br><span class="text-[8px] text-slate-500 font-black uppercase">📍 ${i.suc}</span></div>

                    </div>`).join('');

                // 7. INYECTAR TOP 10
                document.getElementById('lista-top').innerHTML = datosReales.top10.map((i, index) => `
                    <div class="flex items-center gap-2 bg-slate-800/30 p-2 rounded border border-slate-700/50">

'text-emerald-400 border-emerald-500/30' : 'text-blue-400 border-blue-500/30'}">📍 ${i.s}</div></div>
                        <div class="text-[10px] font-black text-white">${i.q} <span class="text-[8px] text-slate-500">U</span></div>

                const commonOptions = { responsive: true, maintainAspectRatio: false, plugins: { legend: { display: false } }, scales: { y: { display: false }, x: { grid: { display: false }, ticks: { color: '#475569', font: { size: 9, weight: 'bold' } } } } };
                
                if(chartSemanal) chartSemanal.destroy();
                chartSemanal = new Chart(document.getElementById('graficaSemanal').getContext('2d'), { type: 'bar', data: { labels: ['L', 'M', 'M', 'J', 'V', 'S', 'D'], datasets: [{ data: datosReales.graficas.semanal, backgroundColor: '#10b981', borderRadius: 4 }] }, options: commonOptions });
                
                if(chartMensual) chartMensual.destroy();
                chartMensual = new Chart(document.getElementById('graficaMensual').getContext('2d'), { type: 'line', data: { labels: ['E', 'F', 'M', 'A', 'M', 'J', 'J', 'A', 'S', 'O', 'N', 'D'], datasets: [{ data: datosReales.graficas.mensual, borderColor: '#3b82f6', tension: 0.4, fill: true, backgroundColor: 'rgba(59, 130, 246, 0.05)', borderWidth: 3 }] }, options: commonOptions });

                // 9. Ocultar loader de forma elegante
                loader.style.opacity = '0';
                setTimeout(() => loader.style.display = 'none', 400);
